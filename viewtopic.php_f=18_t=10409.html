Find this file:
PerfOptions.ini
in:
c:\Documents and Settings\[user\Dokumenter\My Games
and change the: DIFFICULTY_SCALE = 1.00
fx. DIFFICULTY_SCALE = 2.00

this will make the enemies' weapons hit harder and have more hitpoints(i presume)


[PerfVersion
version=5
SkipMachineWarnings=TRUE

[PerfCount
count=36

[PerfOptions
mouse_speed = 0.50
gamma_ramp = 0.30
DitherControl = 1.00
MipFilterControlCallback = 0.00
TurnSensitivity = 0.00
generic_effect_perf = 0.60
Nebula = 0.60
SunEffect = 0.60
Asteroids = 0.60
space_dust_detail = 0.00
star_sphere = 0.60
AtmosphereMaterialConfig = 1.00
HUDAnimConfig = 1.00
level_of_detail = 0.60
use_environment_maps = 1.00
high_detail_textures = 0.60
sound_max_dist_scale = 1.00
sound_level_of_detail = 1.00
USE_3D_SOUNDS = 1.00
audio_music = 1.00
audio_ambient = 1.00
audio_sfx = 1.00
audio_interface = 1.00
audio_voice = 1.00
rtc_performance = 0.60
general_graphics = 0.60
advanced_graphics = 0.00
default_graphics = 0.60
skip_transition_scripts = 1.00
enable_tooltips = 0.00
enable_rollover = 1.00
cockpit = 1.00
USE_DYNAMIC_COMM_HEAD = 1.00
DIFFICULTY_SCALE = 1.00 <------------------------------this one
autohide_maneuver_bar = 0.00
